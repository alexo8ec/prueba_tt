Crear la base por powershell

docker exec -it mssql /opt/mssql-tools18/bin/sqlcmd `
  -S localhost `
  -U sa `
  -P "LaMeg@Password123!" `
  -C `
  -Q "CREATE DATABASE AuthDB"
